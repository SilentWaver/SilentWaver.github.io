#### 实验1  **数据链路层协议的形式化描述与验证**   



```promela
#define MAXSEQ 6
mtype={Msg,Ack,Nak,Err,Miss};
chan SenderToReceiver=[1]of{mtype,byte,byte};
chan ReceiverToSender=[1]of{mtype,byte,byte};

proctype SENDER(chan InCh,OutCh){
    byte SendData;
    byte SendSeq;
    byte ReceivedSeq
    SendData=0;
    SendData=0;
    
    do
    ::skip
again: 
    if
    ::OutCh!Msg(SendData,SendSeq)
    ::OutCh!Err(0,0)
    ::OutCh!Miss(0,0)
    fi; 
    
    if
    ::timeout -> goto again
    ::InCh?Miss(0,0)-> goto again
    ::InCh?Err(0,0)-> goto again
    ::InCh?Nak(ReceivedSeq,0)->end1: goto again
    ::InCh?Ack(ReceivedSeq,0)->   
        if
        ::(ReceivedSeq== SendSeq)->
        SendSeq = (SendSeq+1)%MAXSEQ;
        SendData = (SendData+1)%MAXSEQ;
        ::(ReceivedSeq!=SendSeq)->
        end2: goto again
        fi;
 
    fi;
    od;
 }

 proctype RECEIVER(chan InCh,OutCh){
    byte ReceivedData;
    byte ReceivedSeq;
    byte ExpectedData=0;
    byte ExpectedSeq;
    
    do
    ::InCh?Msg(ReceivedData,ReceivedSeq)->
        if
        ::(ReceivedSeq==ExpectedSeq)->
            progress: 
            ExpectedSeq=1+ExpectedSeq;
            ExpectedData=(ExpectedData+1)%MAXSEQ;
            if
            ::OutCh!Miss(0,0)
            ExpectedSeq=ExpectedSeq-1;
            ExpectedData=(ExpectedData-1)%MAXSEQ;
            ::OutCh!Ack(ReceivedSeq,0)
            ::OutCh!Err(0,0)
            ExpectedSeq=ExpectedSeq-1;
            ExpectedData=(ExpectedData-1)%MAXSEQ;
            fi;
    
        ::(ReceivedSeq!=ExpectedSeq)
            if
            ::OutCh!Nak(ReceivedSeq,0)
            ::OutCh!Err(0,0)
            ::OutCh!Miss(0,0)
            fi;
    
        fi;
    
    ::InCh?Err(0,0)
        if
        ::OutCh!Nak(ReceivedSeq,0)
        ::OutCh!Err(0,0)
        ::OutCh!Miss(0,0)
        fi;
    
    ::InCh?Miss(0,0)->skip;
    od;
 }
 
 init{
    atomic{
        run SENDER(ReceiverToSender,SenderToReceiver);
        run RECEIVER(SenderToReceiver,ReceiverToSender);
    }
 }

```

### 实验2          **AB协议的形式化描述与验证**

```
mtype = {Err,a,b};

chan SenderToReceiver = [1] of {mtype,byte};
chan ReceiverToSender = [1] of {mtype,byte};

proctype A(chan InCh, OutCh)
{


S5: 
   if 
   ::OutCh!a(0)     
   ::OutCh!Err(0)
   ::OutCh!a(0)
   ::OutCh!a(0)
   ::OutCh!a(0)
   fi;

   goto S4;

S4:
   if
   ::InCh?Err(0)->goto S5
   ::InCh?b(1)-> goto S1
	::InCh?b(0)-> goto S1
   fi;

S3:
   if      
   ::OutCh!Err(0)
   ::OutCh!a(1)
   ::OutCh!a(1)
   ::OutCh!a(1)
   ::OutCh!a(1)
   
   fi;
   goto S2;

S2:
   if
   ::InCh?Err(0)->goto S5
   ::InCh?b(1)->goto S1
   ::InCh?b(0)->goto S3
   fi;

S1:        
	if 
   ::OutCh!a(1)
   ::OutCh!Err(0)     
   ::OutCh!a(1)
   ::OutCh!a(1)
   ::OutCh!a(1)
   fi;
	goto S2   

}

proctype B(chan InCh, OutCh){
   
   goto S2;

S1:
   if
   ::OutCh!Err(0)
   ::OutCh!b(1)
   ::OutCh!b(1)
   ::OutCh!b(1)
   ::OutCh!b(1)
   fi;
   goto S2;

S2:
   if
   ::InCh?Err(0)->goto S5
   ::InCh?a(0)->goto S3

S3:
   if
   ::OutCh!Err(0)
   ::OutCh!b(1)
   ::OutCh!b(1)
   ::OutCh!b(1)
   ::OutCh!b(1)

   goto S2;

S4:
   if
   ::InCh?Err(0)->goto S5
   ::InCh?a(0)->goto S1
   ::InCh?a(1)->goto S1
   fi;
S5:    
   if
   ::OutCh?Err(0)
   ::OutCh!b(0)
   ::OutCh!b(0)
   ::OutCh!b(0)
   ::OutCh!b(0)
   fi;
   goto S4

}

   atomic{
	   run A(ReceiverToSender,SenderToReceiver);
```

### 实验3          **GBN协议的形式化描述与验证**  

```
#define MAX 25 //定义发送报文计数的最大值

chan s_r=[10] of {mtype,byte,byte};
chan r_s=[10] of {mtype,byte,byte};

mtype={mesg, ack, err};

proctype udt_sender() //发送端进程
{
  byte s,r,swl;/*swl：first identify s:next trans r：ackNO*/
  swl = 0;
  do   
    ::swl = swl;
    progress:
        s = swl;
    progress1:   /*随机发送多个分组*/
        if
        ::s_r!mesg(0,s)->
            (swl<=s)->
            s = (s+1)%MAX;
            if
            ::goto progress1;
            ::skip
            fi;

        ::s_r!err(s,0) ->
            (swl<=s)->
            s = (s+1)%MAX;
            if
            ::goto progress1;
            ::skip
            fi;

        ::skip -> //丢失
            (swl<=s)->s = (s+1)%MAX;
            if
            ::goto progress1;
            ::skip
            fi;
        fi;

        if
        ::timeout ->goto progress/*超时重发*/
        ::r_s?err(0,r) ->skip   /*错误丢弃*/
        ::r_s?ack(r,0) ->       
            if
            ::(r<swl) -> skip
            ::(r>s) ->skip
            ::(swl<=r<=s) ->
                swl = r;
                goto progress;
            fi;
        fi;
  od
}

proctype udt_receiver() //接收端进程
{
    byte t,es;/*t：接收数据，es：预期接收数据*/
    es = 0;
    do
        ::s_r?mesg(0,t) ->
            if
            ::(t==es)->
                progress2:es = (es + 1)%MAX;
                if
                ::r_s!ack(es,0)
                ::r_s!err(0,es)
                ::skip
                fi

            ::(t!=es)->
                if
                ::r_s!ack(es,0)
                ::r_s!err(0,es)
                ::skip
                fi
            fi

        ::s_r?err(t,0)->
            if
            ::r_s!ack(es,0)
            ::r_s!err(0,es)
            ::skip
            fi
    od
}


init
{
  run udt_sender();
  run udt_receiver();
}
```
